cmake_minimum_required(VERSION 3.0.0)
project(tcp-server VERSION 0.1.0)

SET(CMAKE_BUILD_TYPE "RELEASE") 
SET(CMAKE_C_FLAGS_DEBUG "$ENV{CFLAGS} -O0 -Wall -g -ggdb")
SET(CMAKE_C_FLAGS_RELEASE "$ENV{CFLAGS} -O3 -Wall")

link_libraries(hiredis)
add_subdirectory(md5)
add_subdirectory(file)
add_subdirectory(str)
add_subdirectory(log)
add_subdirectory(network)

add_executable(client client.c)
add_executable(server server.c)
target_link_libraries(client md5)
target_link_libraries(client file_process)
target_link_libraries(client net)
target_link_libraries(client str)
target_link_libraries(client log)
target_link_libraries(server md5)
target_link_libraries(server file_process)
target_link_libraries(server net)
target_link_libraries(server str)
target_link_libraries(server log)
